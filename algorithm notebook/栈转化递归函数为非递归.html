1 问题

利用一个栈实现一下递归函数的非递归计算

\[ P_n(x)=\left\{ \begin{array}{lcl} 1 & & {n = 0}\\ 2x & & {n = 1}\\ 2xP_{n-1}(x) - 2(n-1)P_{n-2}(x) & & {n > 1} \end{array} \right. \]

2 递归

double fun(int n, double x)
{
    if(n == 0)
        return 1;
    else if(n == 1)
        return 2*x;
    else
        return 2*x*fun(n-1, x) - 2*(n-1)*fun(n-2, x);
}

3 非递归

#define N 1024
double fun(int n, double x)
{
    struct stack
    {
        int n;
        double p;
    }st[N];
    int top = -1;
    double p1 = 1, p2 = 2*x;
    for(int i = n; i >= 2; --i)
        st[++top].n = i;
    while(top >= 0)
    {
        st[top].p = 2*x*p2 - 2*(st[top].n-1)*p1;
        p1 = p2;
        p2 = st[top].p;
        --top;
    }
    return p2;
}
